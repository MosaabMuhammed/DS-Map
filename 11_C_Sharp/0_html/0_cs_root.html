    Grouping with dictionaries 
    names = ['mosaab', 'cool', 'hany', 'shady', 'muhamed', 'raymond']

# Basic
d = {}
for name in names:
    key = len(name)
    if key not in d:
        d[key] = []
    d[key].append(name)

# Intermediate
d = {}
for name in names:
    key = len(name)
    d.setdefault(key, []).append(name)

# Pythonic
from collections import defaultdict

d = defaultdict(list)
for name in names:
    key = len(name)
    d[key].append(name)
