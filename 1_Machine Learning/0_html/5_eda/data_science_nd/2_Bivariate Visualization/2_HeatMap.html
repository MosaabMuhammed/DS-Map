Annotate the Heat Map¶

In [35]:
# hist2d returns a number of different variables, including an array of
# counts
plt.figure(figsize=(12, 6));
bins_x = np.arange(.3, 7+.5, .5)
bins_y = np.arange(6, 58+2.5, 2.5)
h2d = plt.hist2d(data=df, x='displ', y='comb',
                bins=[bins_x, bins_y], cmap='viridis_r', cmin=.5);

# Annotation Code
counts = h2d[0]
colors = ['white', 'black']

# Loop through the cell counts and add text annotations for each.
for i in range(counts.shape[0]):
    for j in range(counts.shape[1]):
        c = counts[i, j]
        
        if c >= 7: # increase visibility on darkest cells
            plt.text(bins_x[i]+0, bins_y[j]+0, int(c),
                     ha = 'center', va = 'center', color = 'white')
        elif c > 0:
            plt.text(bins_x[i]+0, bins_y[j]+0, int(c),
                     ha = 'center', va = 'center', color = 'black')

plt.colorbar();
